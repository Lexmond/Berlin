##----------------------------- Install -------------------------------##

# Create new environment
conda create --name nextalign

# Activate Nextalign environment
conda activate nextalign   

# Install Nextalign
conda install -c bioconda nextalign

# Install Snakemake
conda install -c bioconda snakemake

# Deactivate conda
conda deactivate

# Help files
nextalign --help
nextalign run --help

# Run Snakefile
Snakemake --cores 'all'
